In .bashrc, set the branch to appear with font colour:

gitColor() {
  GIT_LOCATION=`__git_ps1 "%s"`
  GIT_COLOR="37m"
  if [ ! $GIT_LOCATION == "master" ];
  then
    GIT_COLOR="31m";
  fi
  echo $GIT_COLOR
}
PS1='\[\033[01;34m\]\w\[\e[0;$(gitColor)\]$(__git_ps1 " (%s)")\[\e[0;34m\] \$\[\033[00m\] '
